\begin{minted}{batch}

problem.exe
std.exe

\subsection*{Linux}
\begin{minted}{bash}
#!/bin/bash
while true; do
    ./data > data.in && ./std < data.in > std.out && ./problem < data.in > problem.out
    if diff std.out problem.out; then printf "Accepted\n"
    else
        printf "Wrong Answer\n"
        exit 0
    fi
done
\end{minted}
